# npm install # to install stuff from the gulpfile.js



# gulp tutorial

$ npm install -g gulp

$ npm init # generate a `package.json` file

$ npm install gulp  --save-dev  # installs the `gulp` command

$ touch gulpfile.js  # generate the gulpfile.js file

$ npm install gulp-minify-css --save-dev

$ npm install gulp-concat --save-dev
